\frame{
	\frametitle{1G in Germany}
	
	\textbf{No-Success factors of the C-Net}
	
	\begin{itemize}
		\item It was proprietary, a closed system
		\item It was expensive
		\begin{itemize}
			\item If amount of users reaches capacity limits, prices were raised
			\item In contrast: NMT fixed their technology to cope the demand
		\end{itemize}
		\item Only few attracted users (No positive externalities)
	\end{itemize}
}

\frame{
	\frametitle{1G in Germany}
	
	\textbf{Reasons for the disappointing success}
	
	\begin{itemize}
			\item The lack of competition in Germany
			\item The lack of leadership (regarding government as well as other companies)
			\item The closed system caused a monopoly for infrastructure and handsets
			\item This also prevented prices from falling
	\end{itemize}
}
